Find the Smaller Angle

        var Minimal_Angle = (h, m) => 
        {
            if (h==12) {
                h=0;
            }
            if (m==60) {
                m=0;
                h += 1;
                if (h>12) {
                    h -= 12;
                }
            }
            let h_angle = 0.5 * (h * 60 + m);
            let m_angle = 6 * m;
            let angle = Math.abs(h_angle - m_angle);
            angle = Math.min(360 - angle, angle);
            return angle;
        };

Perfect Number Check.

        var Perfect_Check = (N) => 
        {
            let temp=0;
            for (let i=1; i<=N; i++) {
                if (N%i===0) {
                    temp += i;
                }
            }
            if (temp === N && temp !==0) {
                return "YES";
            }
            else {
                return "NO";
            }
        };

Reverse a Number.

        var Reverse_Number = (N) => 
        {
            let lastdigit=0;
            let reverse=0;
            while (N !== 0) {
                lastdigit=N%10;
                reverse=(reverse*10)+lastdigit;
                N=Math.floor(N/10);
            }
            return reverse;
        };
